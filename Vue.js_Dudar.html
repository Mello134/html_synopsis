  • 1. Что такое Vue.js?

    1. Vue.js - фреймворк, для создания динаминых web сайтов, хорошо подходит для SPA сайтов.
      Например нажатие кнопки - но перез перезагрузки всей страницы (Лайк, рейтинг, фильтры, листание картинок, комментарии)
    2. Vue.js - облегчет создание сайтов и является прямым конкурентом библиотеки jQuery
    3. Для установки Vue.js, необходимо перейти на официальный сайт https://vuejs.org/, и скопировать cкрипт в html

    2. Принимаем в поле input число, сразу выводим в html и введённое число + введённое число умноженное на 2 
      // Принимаем в поле input число, 
// сразу выводим в html и введённое число + 
// введённое число умноженное на 2

new Vue ({
	el: '#app',
	data: {
		value: 1, // {{ value }} - по учаолчанию 1
		doubleValue: 1 // {{ doubleValue }} - по учаолчанию 1
	},
	methods: {
		// создаём свою функцию обработчик
		// value - в функции не одно и тоже что и в data, они не пересекаются!
		increment (value) {
			// this.value - это value ил data
			// = value - это value из methods -> increment (т.е то что введут в поле input)
			this.value = value
			this.doubleValue = value * 2 // умножаем value на 2
			// ИЛИ
			// this.doubleValue = this.value * 2 // одно и тоже но так лучше
		}
	}
})

new Vue ({

		value: 1, // {{ value }} - по учаолчанию 1
	},

		// создаём свою функцию обработчик
		// value - в функции не одно и тоже что и в data, они не пересекаются!
		increment (value) {

		// {{ doubleValue }}

		}
	}
})
    4. Во Vue.js - поддерживается обычный синтаксис JS

new Vue ({
	el: '#app',
	data: {
		value: 1, // {{ value }}
	},
	methods: {
		increment (value) {
			this.value = value
			if (value == 25) // если ввели 25
				alert("Ввели число 25"); // всплывающее окно
		}
	},
		computed: {
		doubleValue () { // {{ doubleValue }}
			return this.value * 2
		}
	}
});

      // глоюальный фильтр для всех Vue

// ('своё_название-своего_фильтра', function(параметр_называем_сами) {....})
Vue.filter ('capitalize', function (text) {
	if (!text) return '' // если текста нет, возвратим пустою строку
	text = text.toString() // вводные данные превращаем в строку
	
	var capitalFirst = text.charAt(0).toUpperCase(); // берём первый символ.ВЕРХНИЙ РЕГИСТР()
	var otherSimb = text.slice(1, text.length); // слайс по всем символам кроме первого
	return capitalFirst + otherSimb; // Возвращаем, всю строку, первый символ в Верхрем регистре
});

	<div>
		<!-- :value='msg' - изначально в поле будет значение msg (тот что из props) -->

		<input type="text" :value='msg' @input='changeMess'>

      Пишем логику для метода changeMess - это собственный метод 
      <!-- JS логика Field.vue -->
<script>
	export default {
	  // msg это аттрибут, которые указали в Example.vue внутри теггов app-input
	  props: ['msg'],
	  data () {
	    return {
	    	// изначально mess - пустое поле
	    	mess: ''
	    }
	  },
	  methods: {
	  	// обработчик события changeMess
	  	changeMess (event) {
	  		// к значению mess - Применяем, то значение что ввели в поле Input
	  		this.mess = event.target.value;
	  	}
	  },
	}
</script>
      На данной этапе всё работает, выводится тот текст который в оле, но на данный момент мы не можем менять значение текста, он постоянно переписывается на начальный текст msg
      Всё что делали ранее, это помещали данные из родительского файла в дочерние файлы
    9. Ниже будем помещать данные из дочернего файла в родительские
      В Field.vue - в методе обрадобчика события changeMess - добавляем (дописываем инструкцию): 
      <!-- JS логика Field.vue -->
<script>
	export default {
	  // msg это аттрибут, которые указали в Example.vue внутри теггов app-input
	  props: ['msg'],
	  data () {
	    return {
	    	// изначально mess - пустое поле
	    	mess: ''
	    }
	  },
	  methods: {
	  	// обработчик события changeMess
	  	changeMess (event) {
	  		// к значению mess - Применяем, то значение что ввели в поле Input
	  		this.mess = event.target.value;
	  		// $ - означает что это встроенная функция
	  		// в $emit - прописываем функцию, которую будем передовать в родительский файл ('название функции любое', 'параметр функции, что передаём')
	  		this.$emit('messChange', this.mess);
	  	}
	  },
	}
</script>
    10. В Example.vue - внутри теггов app-input - передаём ту функцию тот параметр, который находится в дочернем файле, Field.vue 
      <!-- шаблон/подшаблон -->

		<h1>{{ message }}</h1>

		<!-- вывожу компонент app-input, см скрипт ниже-->
		<!-- @messCange - это функция из Field.vue -->
		<!-- $event - это то значение которое передали в качестве параметра, так же в Field.vue в скрипте - т.е в данном случае this.mess -->
		<!-- выводим message, присваевая message = то что написал пользователь, скрипт в Field.vue - в дочернем файлу -->
		<app-input :msg="message" @messChange="message = $event"></app-input>

	</div>
